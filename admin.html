function approveGame(index) { const pending = JSON.parse(localStorage.getItem('minibossx_pending')) || []; const game = pending[index]; // Ask for the folder name const folderName = prompt('Enter folder name for this game (e.g., odd-even):', game.name.toLowerCase().replace(/[^a-z0-9]+/g, '-')); if (!folderName) return; // Create download link for the ZIP file if (game.fileData) { // Create a downloadable ZIP file const byteCharacters = atob(game.fileData); const byteNumbers = new Array(byteCharacters.length); for (let i = 0; i < byteCharacters.length; i++) { byteNumbers[i] = byteCharacters.charCodeAt(i); } const byteArray = new Uint8Array(byteNumbers); const blob = new Blob([byteArray], {type: 'application/zip'}); const url = URL.createObjectURL(blob); // Show instructions alert(`✅ Game approved!\n\nNext steps:\n1. Download the ZIP file from the link that will open\n2. Extract to folder: /games/${folderName}/\n3. Upload all files to GitHub\n4. Make sure index.html is in the root\n\nGame will be live at: /games/${folderName}/`); // Open download link window.open(url); } else { alert('No file data found'); return; } // Create approved game const approvedGame = { id: Date.now(), name: game.name, developer: game.developer, description: game.description, category: game.category, platform: game.platform, gameUrl: `/games/${folderName}/`, dateApproved: new Date().toLocaleString() }; // Add to approved const approved = JSON.parse(localStorage.getItem('minibossx_approved')) || []; approved.push(approvedGame); localStorage.setItem('minibossx_approved', JSON.stringify(approved)); // Mark as approved in pending game.status = 'approved'; localStorage.setItem('minibossx_pending', JSON.stringify(pending)); // Reload loadPending(); loadApproved(); }
